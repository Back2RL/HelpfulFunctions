-- Ueb01.2 b)
select mov.title Filmtitel, plays.character Charakter, pers.name Person
from moviedb.movie mov, moviedb.plays plays, moviedb.person pers
where mov.id = plays.movie
and pers.id = plays.player
and (mov.year < 1975 or mov.year > 2006)
order by mov.title;

-- Ueb01.2 c)
select mov.title Filmtitel
from moviedb.movie mov, moviedb.directs dir
where 
dir.movie = mov.id
and not exists(select moviedb.person.id
from moviedb.person
where dir.director = moviedb.person.id);

-- Ueb01.2 d)
select mov.title Filmtitel
from moviedb.movie mov
join moviedb.rating rat on mov.id = rat.movie
join moviedb.directs dir on mov.id = dir.movie
join moviedb.person pers on dir.director = pers.name;
